from IPython.core.magic import register_line_magic, register_cell_magic

@register_line_magic
def lmagic(line):
    "my line magic"
    return line

@register_cell_magic
def cmagic(line, cell):
    "my cell magic"
    return line, cell

# We delete to avoid name conflicts for automagic to work
del lmagic
